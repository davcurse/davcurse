cmp R2, R4          ;Compares R2 and R4 for looping purposes
je 46               ;If compare is true, jump to instruction 46 to end program
inc R2              ;Increments R2 for looping purposes
add R0, R1          ;Add R0 and R1 (previous 2 numbers in sequence)
mov R5, R0          ;Move sum of R0 and R1 to R5
mov [R2], R5        ;Store R5 (sum of R0 and R1) into memory
